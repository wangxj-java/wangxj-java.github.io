Vue.js

无标签
发布日期:   2020-07-02
阅读次数:  

Vue.js（读音 /vjuː/, 类似于 view） 是一套构建用户界面的渐进式框架。
Vue 只关注视图层， 采用自底向上增量开发的设计。
Vue 的目标是通过尽可能简单的 API 实现响应的数据绑定和组合的视图组件。

1 vue.js安装

我们可以在 Vue.js 的官网上直接下载 并且用script标签引入

https://cdn.jsdelivr.net/npm/vue/dist/vue.js

2 基础语法

<!DOCTYPE html>
<html lang="en" xmlns:v-if="http://www.w3.org/1999/xhtml" xmlns:v-for="http://www.w3.org/1999/xhtml"
      xmlns:v-on="http://www.w3.org/1999/xhtml" xmlns:v-bind="http://www.w3.org/1999/xhtml">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
    <script src="https://unpkg.com/vue/dist/vue.js"></script>
    <script src="https://cdn.bootcdn.net/ajax/libs/axios/0.19.2/axios.min.js"></script>
</head>
<body>
    <div id="app">
        <!--文本赋值-->
        <p>{{message}}</p>
        <!--判断-->
        <p v-if="ok">yes</p>
        <p v-else>no</p>

        <!--遍历-->
        <li v-for="item in items">
            {{item.username}}
        </li>

        <!--事件绑定-->
        <p>
            <input type="button" value="点击过新年" v-on:click="hello('亲爱的，新年好呀')"/>
        </p>

        <!--双向绑定-->
        <p>
            <input type="text" v-model="age">{{age}}
        </p>
        <!--组件 component-->
        <p>
            <myli v-for="item in items" v-bind:i="item.username"></myli>
        </p>

        <!--钩子函数-->
        <!--<p>
            {{info.name}}
        </p>
        <p>
            <a v-bind:href="info.url">点击</a>
        </p>-->

    </div>

    <script>

        Vue.component("myli",{
            props: ["i"],
            template: '<li>{{i}}</li>'
        });

        var vm = new Vue({
            el: "#app",
            data: {message:"王秀健",
                ok: true,
                items: [{username: "张三"},
                    {username: "李四"},
                    {username: "王五"},
                ],
                age: ""
            },
            methods: {
                hello: function (message) {
                    alert(message);
                }
            },
            /*mounted(){//钩子函数，链式编程
                axios.get('data.json').then(Response=>(this.info=Response.data ));

            },*/

            /*data(){
                return{

                        name: null,

            }*/

        });

    </script>
</body>
</html>

3 插槽组件

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
    <script src="https://unpkg.com/vue/dist/vue.js"></script>
    <script src="https://cdn.bootcdn.net/ajax/libs/axios/0.19.2/axios.min.js"></script>
</head>
<body>

    <div id="app">
        <!--<li v-for="item in items"><span>{{item}}</span></li>-->
        <!--自定义插槽-->
        <todo>
            <todo_ti  slot="todo_ti" :title="title"></todo_ti>
            <todo_li slot="todo_li" v-for="i in items" :item="i"></todo_li>

        </todo>

    </div>

    <script>

        Vue.component("todo",{
            template: '<div>\
                           <slot name="todo_ti"></slot>\
                           <ul>\
                                <slot name="todo_li"></slot>\
                           </ul>\
                        </div>'
        })
        /*定义title插槽*/
        Vue.component("todo_ti",{
            props: ['title'],
            template: '<div>{{title}}</div>'
        })
        /*定义li标签插槽*/
        Vue.component("todo_li",{
            props: ['item'],
            template: '<li>{{item}}</li>'
        })
        /*注册vue组件需要在初始化vue应用之前！！！*/
        var vm = new Vue({
            el: "#app" ,
            data: {
                title: "列表标题",
                items: ["java","php"]
            }
        });

    </script>

</body>
</html>

4 自定义事件内容分发

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
    <script src="https://unpkg.com/vue/dist/vue.js"></script>
    <script src="https://cdn.bootcdn.net/ajax/libs/axios/0.19.2/axios.min.js"></script>
</head>
<body>

    <div id="app">
        <!--<li v-for="item in items"><span>{{item}}</span></li>-->
        <!--自定义插槽-->
        <todo>
            <todo_ti  slot="todo_ti" :title="title"></todo_ti>
            <todo_li slot="todo_li" v-for="(i,index) in items" :item="i"
                  :index="index"  v-on:remove="removeData(index)" :key="index"></todo_li>

        </todo>

    </div>

    <script>

        Vue.component("todo",{
            template: '<div>\
                           <slot name="todo_ti"></slot>\
                           <ul>\
                                <slot name="todo_li"></slot>\
                           </ul>\
                        </div>'
        })
        /*定义title插槽*/
        Vue.component("todo_ti",{
            props: ['title'],
            template: '<div>{{title}}</div>'
        })
        /*定义li标签插槽*/
        Vue.component("todo_li",{
            props: ['item','index'],
            template: '<li>{{index}}-----{{item}}<button @click="remove">删除</button></li>' ,
            methods: {
                remove: function (index) {
                    this.$emit('remove',index);
                }
            }
        })
        /*注册vue组件需要在初始化vue应用之前！！！*/
        var vm = new Vue({
            el: "#app" ,
            data: {
                title: "列表标题",
                items: ["java","php","spring","boot"]
            },
            methods: {
                removeData: function (index) {
                    this.items.splice(index,1) ;//指定元素删除
                }
            }
        });

    </script>

</body>
</html>
